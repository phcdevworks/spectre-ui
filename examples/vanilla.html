Recipe API

TypeScript class recipes shown with their CSS output.

// Buttons
getButtonClasses({ variant: "primary", size: "lg" })
// → "sp-btn sp-btn--primary sp-btn--lg"

getButtonClasses({ variant: "cta", pill: true, fullWidth: true })
// → "sp-btn sp-btn--cta sp-btn--md sp-btn--pill sp-btn--full"

// Badges
getBadgeClasses({ variant: "warning", size: "sm" })
// → "sp-badge sp-badge--warning sp-badge--sm"

// Icon Boxes
getIconBoxClasses({ variant: "info", size: "lg" })
// → "sp-iconbox sp-iconbox--info sp-iconbox--lg"

// Cards
getCardClasses({ variant: "outline", interactive: true, padded: true })
// → "sp-card sp-card--outline sp-card--interactive sp-card--padded"

// Inputs
getInputClasses({ state: "error", size: "lg" })
// → "sp-input sp-input--lg sp-input--error"

// Pricing Cards
getPricingCardClasses({ featured: true })
// → "sp-pricing-card sp-pricing-card--featured"

// Ratings
getRatingStarClasses(true)
// → "sp-rating-star sp-rating-star--filled"
